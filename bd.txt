create table conta(
    idconta serial not null,
    nome varchar(50) not null,
    saldo numeric(10,1),
    senha varchar(6) not null,
    constraint fk_conta primary key (idconta)
)	

insert into conta(nome,saldo,senha) values('Murillo',100,'1234');
